  private handleDeleteClick() {
    const dialog = this.shadowRoot?.querySelector('delete-dialog') as any;
    dialog?.open();
  }
 
  private async handleDeleteConfirmed(event: CustomEvent) {
    const { employeeId } = event.detail;
    if (employeeId) {
      this.dispatchEvent(new CustomEvent('employee-deleted', {
        detail: { employeeId },
        bubbles: true,
        composed: true
      }));
    }
  }
 
  private handleEditClick() {
    window.location.href = `/edit-employee.html?id=${this.employee.id}`;
  }
